#Projeto 1 - Ouvidoria ABC
listaProtocolos = []
listaProtocolosArmazenar = []
listaProtocolosExibir = []
manifestacao = []
listaManifestacaoExibir = []
listaReclamacao = []
listaSugestao = []
listaElogio = []
tipo = 0
manifestacaoQuebrada = []
acharProtocolo = []
opcao = 0
while opcao != 7:
    print('===========================')
    print('Ouvidoria Universidade ABC')
    print('===========================')
    # lendo a variável opcao, que irá direcionar o usuário de acordo com a ação escolhida no menu
    opcao = int(input('1 - Listar todas as manifestações \n'
                      '2 - Listar todas as sugestões \n'
                      '3 - Listar todas as reclamações \n'
                      '4 - Listar todos os elogios \n'
                      '5 - Enviar uma nova manifestação \n'
                      '6 - Pesquisar protocolo por número (ID)\n'
                      '7 - Sair'
                      '\nDigite um número de 1 a 7 para a ação desejada:\n'))
    # certificando que o usuário digite uma opção válida, entre 1 e 7
    while opcao < 1 or opcao > 7:
        print('\nEntrada inválida!')
        opcao = int(input('1 - Listar todas as manifestações \n'
                          '2 - Listar todas as sugestões \n'
                          '3 - Listar todas as reclamações \n'
                          '4 - Listar todos os elogios \n'
                          '5 - Enviar uma nova manifestação \n'
                          '6 - Pesquisar protocolo por número (ID)\n'
                          '7 - Sair'
                          '\nDigite um número de 1 a 7 para a ação desejada:\n'))

    #Menu: 1 -  Listar todas as manifestações
    if opcao == 1:
        print('---------------------------------')
        print('Lista com todas as manifestações: ')
        print('---------------------------------\n')
        for i in range(len(listaProtocolosArmazenar)):
            print(listaProtocolosArmazenar[i])
        if len(listaProtocolosArmazenar) == 0:
            print("Ainda não foram registradas manifestações. Digite a opção 5 no menu principal para enviar uma nova manifestação!\n")

    #Menu: 2 -  Listar todas as sugestões
    if opcao == 2:
        print('-----------------------------')
        print('Lista com todas as sugestões:')
        print('-----------------------------\n')
        for i in range(len(listaSugestao)):
            print(listaSugestao[i])
        if len(listaSugestao) == 0:
            print("Ainda não foram registradas sugestões. Digite a opção 5 no menu principal para enviar uma sugestão!\n")

    #Menu: 3 -  Listar todas as reclamações
    if opcao == 3:
        print('-------------------------------')
        print('Lista com todas as reclamações:')
        print('-------------------------------\n')
        for i in range(len(listaReclamacao)):
            print(listaReclamacao[i])
        if len(listaReclamacao) == 0:
            print("Ainda não foram registradas reclamações. Digite a opção 5 no menu principal para enviar uma reclamação!\n")

    #Menu: 4 -  Listar todos os elogios
    if opcao == 4:
        print('---------------------------')
        print('Lista com todos os elogios:')
        print('---------------------------\n')
        for i in range(len(listaElogio)):
            print(listaElogio[i])
        if len(listaElogio) == 0:
            print("Ainda não foram registrados elogios. Digite a opção 5 no menu principal para enviar um elogio!\n")

    #Menu: 5 - Criar uma nova manifestação
    if opcao == 5:
        nome = input('Digite o nome do requisitante: ')
        tipo = int(input('Escolha o tipo da manifestação:\n'
                         '1 - Reclamação\n'
                         '2 - Sugestão\n'
                         '3 - Elogio\n'))
        #certificando que o usuário escolha um tipo válido
        while tipo < 1 or tipo > 3:
            print('\nEntrada inválida!\n')
            tipo = int((input('Escolha o tipo da manifestação:\n'
                             '1 - Reclamação\n'
                             '2 - Sugestão\n'
                             '3 - Elogio\n')))
        if tipo == 1:
            tipo = 'Reclamação'
        elif tipo == 2:
            tipo = 'Sugestão'
        elif tipo == 3:
            tipo = 'Elogio'
        descricao = input('Digite a descrição da manifestação: ')
        print('\nManifestação enviada com sucesso!\n')

        protocolo = len(listaProtocolos) + 1
        protocoloStr = str(protocolo)
        acharProtocolo.append(protocoloStr)
        manifestacao = protocoloStr + '#' + nome + '#' + tipo + '#' + descricao
        manifestacaoQuebrada = manifestacao.split('#')
        listaProtocolos.append(manifestacaoQuebrada)

        listaProtocolosExibir = 'ID do protocolo: ' + protocoloStr + ' - Requisitante: ' + nome + ' - Tipo: ' + tipo + ' - Descrição: ' + descricao
        listaProtocolosArmazenar.append(listaProtocolosExibir)
        listaManifestacaoExibir = 'ID do protocolo: ' + protocoloStr + ' - Requisitante: ' + nome + ' - Descrição: ' + descricao

        if tipo == 'Reclamação':
            listaReclamacao.append(listaManifestacaoExibir)
        elif tipo == 'Sugestão':
            listaSugestao.append(listaManifestacaoExibir)
        elif tipo == 'Elogio':
            listaElogio.append(listaManifestacaoExibir)

    #Menu: 6 -  Pesquisar uma manifestação pelo número do protocolo
    if opcao == 6:
        protocoloDesejado = int(input('Informe o número do protocolo que deseja pesquisar: '))
        achou = False
        for i in range(0, len(listaProtocolos) + 1, 1):
            if protocoloDesejado == i + 1:
                achou = True
                print('--------------------------')
                print('O protocolo número {} é o:'.format(protocoloDesejado))
                print('--------------------------')
                print('Código: ', listaProtocolos[i][0])
                print('Nome do requisitante: ', listaProtocolos[i][1])
                print('Tipo de manifestação: ', listaProtocolos[i][2])
                print('Descrição: ', listaProtocolos[i][3])
                print()
                break

        if not achou:
            print('------------------------------')
            print('Não há protocolos com esse ID!')
            print('------------------------------')
